  • -1,00 = Correlaci√≥n negativa perfecta
  • -0,90 = Correlaci√≥n negativa muy fuerte
  • -0,75 = Correlaci√≥n negativa considerable
  • -0,50 = Correlaci√≥n negativa media
  • -0,10 = Correlaci√≥n negativa d√©bil
  • 0,00 = No existe correlaci√≥n lineal alguna entre las variables
  • 0,10 = Correlaci√≥n positiva d√©bil
  • 0,50 = Correlaci√≥n positiva media
  • 0,75 = Correlaci√≥n positiva considerable
  • 0,90 = Correlaci√≥n positiva muy fuerte
  • 1,00 = Correlaci√≥n positiva perfecta
